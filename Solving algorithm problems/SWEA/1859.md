``` bash
import sys
sys.stdin = open('input.txt', 'r')
def my_max(X):
    m_max=0
    for i in range(len(X)):
        if m_max<X[i]:
            m_max = X[i]
    return m_max


T= int(input())
for tc in range(1,T+1):
    N=int(input())
    A=list(map(int, input().split()))
    answer=0
    B= my_max(A)
    C =[]
    x=0
    while len(A)!=0 or len(A)!=x:
        try:
            if A[x] ==B:
                del A[:x+1]
                B = my_max(A)
                x=0
            elif A[x] <=B:
                answer+= B-A[x]
                x+=1
        except IndexError:
            pass
    print(f'#{tc} {answer}')
```
