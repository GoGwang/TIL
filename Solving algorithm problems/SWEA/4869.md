# 종이 붙이기
``` bash
import sys
sys.stdin = open('input.txt', 'r')

def paper(X):
    if X%10==0:
        if X==10:
            return 1
        elif X==20:
            return 3
        else:
            return paper(X-10)+(2*paper(X-20))
    else:
        print("10의 배수가 아닙니다.")
'''
X_1=10일 경우 answer = 1
X_2=20일 경우 answer = 3
X_3=30일 경우 answer = 5
X_4=40일 경우 answer = 11
X_5=50일 경우 answer = 21
이므로 
X_n = (X_n-1) +(2* X_n-2) 
'''


T= int(input())
for tc in range(1,T+1):
    N=int(input())
    answer = paper(N)
    print(f'#{tc} {answer}')
```
