# Flatten
``` bash
def my_sort(X): # 오름차순 정렬 함수 정의
    for q in range(len(X)):
        for w in range(q + 1, len(X)):
            if X[q] > X[w]:
                X[q], X[w] = X[w], X[q]


for i in range(1, 11):
    A = int(input())  # 덤프 횟수
    B = list(map(int, input().split())) # 박스들의 높이
    my_sort(B) # 리스트 B를 오름차순으로 정렬

    for j in range(A):
        B[0] += 1 # 가장 높이가 낮은 박스 부분에 +1
        B[-1] -= 1 # 가장 높이가 높은 박스에 -1
        my_sort(B) # 다시 오름차순 정렬

    print(f'#{i} {B[-1] - B[0]}')
```
